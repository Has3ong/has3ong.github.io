def solution():
    S = str(input())

    while True:

        if S[0:2] == "pi":
            S = S[2:]
            continue
        elif S[0:2] == "ka":
            S = S[2:]
            continue
        elif S[0:3] == "chu":
            S = S[3:]
            continue

        if len(S)==0:
            print("YES")
            return
        else:
            print("NO")
            return
solution()
