```python
import sys

def solution():
    N, M = map(int, sys.stdin.readline().split())
    hash = {}
    for _ in range(N):
        site, alphabet = map(str, sys.stdin.readline().split())
        hash[site] = alphabet

    for _ in range(M):
        find = sys.stdin.readline().strip()
        print(hash[find])
solution()
```
