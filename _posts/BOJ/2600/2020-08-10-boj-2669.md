```python
import sys

def solution():
    Field = [[0 for _ in range(101)] for _ in range(101)]
    for _ in range(4):
        x, y, dx, dy = map(int, sys.stdin.readline().split())

        for i in range(x, dx):
            for j in range(y, dy):
                Field[i][j] = 1

    ret = 0
    for f in Field:
        ret += f.count(1)
    print(ret)

solution()
```
