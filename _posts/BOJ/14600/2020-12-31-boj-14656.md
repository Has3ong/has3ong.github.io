```python
import sys

def solution():
    N = int(sys.stdin.readline())
    arr = list(map(int, sys.stdin.readline().split()))

    ret = 0
    for i in range(1, N+1):
        if arr[i-1] != i:
            ret += 1

    print(ret)

solution()
```
