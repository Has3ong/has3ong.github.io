```python
import sys

def solution():
    N = int(sys.stdin.readline())

    arr1 = list(map(int, sys.stdin.readline().split()))
    arr2 = list(map(int, sys.stdin.readline().split()))

    ret = 0
    for i in range(N):
        ret += abs(arr1[i])
        ret += abs(arr2[i])

    print(ret)

solution()
```
