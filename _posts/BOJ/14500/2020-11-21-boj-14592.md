```python
import sys

def solution():
    T = int(input())
    arr = []
    for i in range(T):
        arr.append([i, list(map(int, sys.stdin.readline().split()))])

    arr.sort(key=lambda x:(-x[1][0], x[1][1], x[1][2]))

    print(arr[0][0] + 1)
solution()
```
