```python
import sys

def Check(data):
    data.sort()
    for i in range(len(data) - 1):
        if data[i] in data[i + 1]: 
            return False
    return True

def solution():    
    T = int(input())    
    
    ret=[]

    for _ in range(T):
        data=[]
        
        N = int(input())
        for _ in range(N):
            data.append(sys.stdin.readline().strip())
        ret.append(Check(data))

    for i in ret:
        if i:
            print('YES')
        else:
            print('NO')

solution()
```
