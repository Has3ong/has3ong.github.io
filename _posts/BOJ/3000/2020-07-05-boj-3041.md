```python
def solution():
    ret = 0
    for i in range(4):
        Puzzle = list(input())
        for j in range(4):
            if Puzzle[j] != '.':
                ret += abs((ord(Puzzle[j]) - ord('A')) % 4 - j)
                ret += abs((ord(Puzzle[j]) - ord('A')) // 4 - i)
    print(ret)

solution()
```
