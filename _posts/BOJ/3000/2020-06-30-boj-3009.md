```python
def solution():
    dataX = []
    dataY = []
    for _ in range(3):
        x, y = map(int, input().split(' '))
        dataX.append(x)
        dataY.append(y)

    for i in range(3):
        if dataX.count(dataX[i]) == 1:
            x = dataX[i]
        if dataY.count(dataY[i]) == 1:
            y = dataY[i]
    print(x, y)
solution()
```
