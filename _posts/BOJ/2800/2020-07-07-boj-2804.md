```python
def solution():
    A, B = map(str, input().split(' '))

    Table = [['.' for _ in range(len(A))] for _ in range(len(B))]

    flag = False

    for posA in range(len(A)):
        for posB in range(len(B)):
            if A[posA] == B[posB]:
                flag = True
                break
        if flag:
            break

    Table[posB] = list(A)
    for i in range(len(B)):

        Table[i][posA] = B[i]

    for t in Table:
        print(*t, sep='')
solution()
```
