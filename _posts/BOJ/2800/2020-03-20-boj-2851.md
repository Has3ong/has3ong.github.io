```python
def solution():
    
    data = []
    for _ in range(10):
        data.append(int(input()))

    score = 0
    for i in data:
        score += i
        if score >= 100:
            if score - 100 > 100 - (score - i):
                score -= i
            break
        
    print(score)

solution()
```
