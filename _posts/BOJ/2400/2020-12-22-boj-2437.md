```python
import sys

def solution():
    N = int(sys.stdin.readline())
    arr = list(map(int, sys.stdin.readline().split()))

    arr.sort()
    sum = 0
    for i in range(N):
        if sum + 2 <= arr[i]: break
        sum += arr[i]
    print(sum + 1)

solution()
```
