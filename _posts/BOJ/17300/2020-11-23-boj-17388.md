```python
import sys

def solution():
    S, K, H = map(int, sys.stdin.readline().split())
    if S + K + H >= 100:
        print("OK")
        return
    else:
        temp = min(S, K)
        temp = min(temp, H)

        if S == temp:
            print("Soongsil")
        elif K == temp:
            print("Korea")
        else:
            print("Hanyang")
        return

solution()
```
