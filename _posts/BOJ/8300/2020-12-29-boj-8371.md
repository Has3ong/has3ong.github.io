```python
import sys

def solution():
    N = int(sys.stdin.readline())
    s1 = sys.stdin.readline().strip()
    s2 = sys.stdin.readline().strip()

    ret = 0
    for i in range(N):
        if s1[i] != s2[i]: ret += 1
    print(ret)
solution()
```
