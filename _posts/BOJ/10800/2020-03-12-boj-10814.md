```python
import sys

def solution():
    data = []

    for _ in range(int(sys.stdin.readline())):
        data.append(list(sys.stdin.readline().split())) 

    data.sort(key = lambda x: int(x[0])) 

    for x, y in data:
        print(x, y)

solution()
```
