```python
import sys

def solution():
    S = sys.stdin.readline().strip()
    ans = 0
    count = 0

    for i in range(1, len(S)):
        if S[i-1] ==')' and S[i] == ')': ans += count
        elif S[i-1] == '(' and S[i] == '(': count += 1
    
    print(ans)

solution()
```
