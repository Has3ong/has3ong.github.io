```python
import sys

def solution():
    N = int(input())
    studentList = []
    for _ in range(N):
        studentList.append(list(map(int, sys.stdin.readline().split())))

    for i in studentList:
        rank = 1
        for j in studentList:
            if i[0] < j[0] and i[1] < j[1]:
                rank += 1
        print(rank, end=" ")
solution()
```
