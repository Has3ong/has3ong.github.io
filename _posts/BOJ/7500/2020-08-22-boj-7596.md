```python
def solution():
    cnt = 1
    while True:
        N = int(input())
        if not N:
            break
        array = []

        for i in range(N):
            array.append(str(input()))

        array.sort()
        print(cnt)
        cnt += 1

        for i in array:
            print(i)

solution()
```
