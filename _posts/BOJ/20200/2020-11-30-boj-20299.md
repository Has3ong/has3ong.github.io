```python
import sys

def solution():
    N, K, L = map(int, sys.stdin.readline().split())
    ret = []
    idx = 0
    for _ in range(N):
        x1, x2, x3 = map(int, sys.stdin.readline().split())
        if min(x1, x2, x3) < L or x1 + x2 + x3 < K:
            continue
        idx += 1
        ret.append(x1)
        ret.append(x2)
        ret.append(x3)

    print(idx)
    for r in ret:
        print(r, end=' ')

solution()
```
