```python
def solution():
    N = int(input())

    cnt = 0
    for idx in range(1, N+1):
        for jdx in range(1, idx+1):
            A = idx-jdx + 1
            B = jdx
            cnt += 1
            if cnt == N:
                print(A, B)
                return

solution()
```
