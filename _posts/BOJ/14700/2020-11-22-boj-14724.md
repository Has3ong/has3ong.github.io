```python
def solution():
    N = int(input())
    arr = []
    for _ in range(9):
        arr.append(list(map(int, input().split())))

    ret, idx = 0, 0
    for i in range(9):
        for j in range(N):
            if arr[i][j] > ret:
                idx = i
                ret = arr[i][j]

    dic = ['PROBRAIN', 'GROW', 'ARGOS', 'ADMIN', 'ANT', 'MOTION', 'SPG', 'COMON', 'ALMIGHTY']

    print(dic[idx])
solution()
```
